@name WeaponSeq2
@inputs Button
@outputs Target:entity
@persist Array:array X Y Weapons:array
@trigger 
runOnTick(1)
if(first()){
findIncludePlayerProps(owner())
findByClass("sc_weapon_base")
Weapons = findToArray()
print(Weapons:count())
}
Y = Y + 1
local V = Weapons[Y,entity]
local Wirelink = V:wirelink()
V:setPos(entity():pos()+vec(0,0,10))
Wirelink["Fire",number] = 1
