@name Explosions
@inputs 
@outputs 
@persist 
@trigger 

interval(255)


Bounds = vec(15000,15000,15000)
MinDist = 7000
MaxDist = 15000

while(perf()) {
    Vec = randvec(Bounds,-Bounds)
    if(Vec:distance(entity():pos()) > MinDist && Vec:distance(entity():pos()) && Vec:isInWorld() < MaxDist) {break}
}

fx("sc_shipsplosion_f",Vec,1,ang(),vec(0,0,1),1)
